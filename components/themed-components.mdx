# Themed Components

BioTea uses a theming system built on React Native's `useColorScheme` hook and custom color constants to provide automatic light/dark mode support.

## Theming Architecture

### Color Constants

All theme colors are defined in `constants/Colors.ts`:

```typescript
const tintColorLight = '#0a7ea4';
const tintColorDark = '#fff';

export const Colors = {
  light: {
    text: '#11181C',
    background: '#fff',
    tint: tintColorLight,
    icon: '#687076',
    tabIconDefault: '#687076',
    tabIconSelected: tintColorLight,
  },
  dark: {
    text: '#ECEDEE',
    background: '#151718',
    tint: tintColorDark,
    icon: '#9BA1A6',
    tabIconDefault: '#9BA1A6',
    tabIconSelected: tintColorDark,
  },
};
```

### useThemeColor Hook

The `useThemeColor` hook (`hooks/useThemeColor.ts`) provides the theming logic:

```typescript
import { useColorScheme } from 'react-native';
import { Colors } from '@/constants/Colors';

export function useThemeColor(
  props: { light?: string; dark?: string },
  colorName: keyof typeof Colors.light & keyof typeof Colors.dark
) {
  const theme = useColorScheme() ?? 'light';
  const colorFromProps = props[theme];

  if (colorFromProps) {
    return colorFromProps;
  } else {
    return Colors[theme][colorName];
  }
}
```

**How it works:**
1. Detects current color scheme (light/dark) from device settings
2. Checks if custom colors were passed via props
3. Falls back to default colors from `Colors` constant
4. Returns the appropriate color for current theme

## ThemedText

<Accordion title="Full Source Code (components/ThemedText.tsx:1-61)">
```typescript
import { Text, type TextProps, StyleSheet } from 'react-native';
import { useThemeColor } from '@/hooks/useThemeColor';

export type ThemedTextProps = TextProps & {
  lightColor?: string;
  darkColor?: string;
  type?: 'default' | 'title' | 'defaultSemiBold' | 'subtitle' | 'link';
};

export function ThemedText({
  style,
  lightColor,
  darkColor,
  type = 'default',
  ...rest
}: ThemedTextProps) {
  const color = useThemeColor({ light: lightColor, dark: darkColor }, 'text');

  return (
    <Text
      style={[
        { color },
        type === 'default' ? styles.default : undefined,
        type === 'title' ? styles.title : undefined,
        type === 'defaultSemiBold' ? styles.defaultSemiBold : undefined,
        type === 'subtitle' ? styles.subtitle : undefined,
        type === 'link' ? styles.link : undefined,
        style,
      ]}
      {...rest}
    />
  );
}

const styles = StyleSheet.create({
  default: {
    fontSize: 16,
    lineHeight: 24,
  },
  defaultSemiBold: {
    fontSize: 16,
    lineHeight: 24,
    fontWeight: '600',
  },
  title: {
    fontSize: 32,
    fontWeight: 'bold',
    lineHeight: 32,
  },
  subtitle: {
    fontSize: 20,
    fontWeight: 'bold',
  },
  link: {
    lineHeight: 30,
    fontSize: 16,
    color: '#0a7ea4',
  },
});
```
</Accordion>

## ThemedView

<Accordion title="Full Source Code (components/ThemedView.tsx:1-15)">
```typescript
import { View, type ViewProps } from 'react-native';
import { useThemeColor } from '@/hooks/useThemeColor';

export type ThemedViewProps = ViewProps & {
  lightColor?: string;
  darkColor?: string;
};

export function ThemedView({ style, lightColor, darkColor, ...otherProps }: ThemedViewProps) {
  const backgroundColor = useThemeColor({ light: lightColor, dark: darkColor }, 'background');

  return <View style={[{ backgroundColor }, style]} {...otherProps} />;
}
```

## Best Practices

### Always Use Themed Components

<Tabs>
  <Tab title="Do">
    ```tsx
    import { ThemedText, ThemedView } from '@/components';
    
    <ThemedView>
      <ThemedText>Properly themed content</ThemedText>
    </ThemedView>
    ```
  </Tab>
  
  <Tab title="Don't">
    ```tsx
    import { View, Text } from 'react-native';
    
    <View style={{ backgroundColor: '#fff' }}>
      <Text style={{ color: '#000' }}>Won't adapt to dark mode</Text>
    </View>
    ```
  </Tab>
</Tabs>

### Override Colors Sparingly

Only use custom `lightColor` and `darkColor` props when you need specific brand colors or semantic colors (success, error, etc.).

```tsx
// Good: Semantic colors
<ThemedView 
  lightColor="#fee" 
  darkColor="#511"
>
  <ThemedText lightColor="#c00" darkColor="#f88">Error message</ThemedText>
</ThemedView>

// Avoid: Unnecessary customization
<ThemedText lightColor="#11181C" darkColor="#ECEDEE">
  This is redundant - these are already the defaults
</ThemedText>
```

### Consistent Typography

Use the predefined `type` prop instead of custom font sizes:

```tsx
// Good
<ThemedText type="title">Page Title</ThemedText>
<ThemedText type="subtitle">Section</ThemedText>

// Avoid
<ThemedText style={{ fontSize: 32, fontWeight: 'bold' }}>Title</ThemedText>
```
